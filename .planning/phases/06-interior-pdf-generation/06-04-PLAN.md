```typescript
export interface MarginSet {
  insideIn: number  // Gutter (binding edge) - larger for more pages
  outsideIn: number // Outside edge margin
  topIn: number
  bottomIn: number
}

export function getMargins(pageCount: number, bleed: BleedSetting): MarginSet {
  // Returns mirror margins that scale with page count
  // - Inside margin increases with page count (0.375" to 0.875")
  // - Outside margins: 0.375" with bleed, 0.25" without bleed
}
```

Key insight: KDP uses "mirror margins" where the inside (gutter) margin is larger than outside margins to account for binding. The inside margin scales with page count because thicker books need more room for the binding.
