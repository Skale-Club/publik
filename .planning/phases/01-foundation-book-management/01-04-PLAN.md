    **File 1: `src/app/(dashboard)/page.tsx`** (Server Component)
    ```typescript
    import Link from "next/link"
    import { Plus } from "lucide-react"
    import { db } from "@/infrastructure/db/client"
    import { books } from "@/infrastructure/db/schema/books"
    import { desc, isNull } from "drizzle-orm"
    import { Header } from "@/components/layout/header"
    import { BookCard } from "@/components/books/book-card"

    export default async function DashboardPage() {
      const allBooks = await db
        .select()
        .from(books)
        .where(isNull(books.deletedAt))
        .orderBy(desc(books.createdAt))

      return (
        <div>
          <div className="flex items-center justify-between">
            <Header title="Books" description="Manage your book projects" />
            <Link
              href="/dashboard/books/new"
              className="inline-flex items-center rounded-md bg-gray-900 px-4 py-2 text-sm font-medium text-white hover:bg-gray-800"
            >
              <Plus className="mr-2 h-4 w-4" />
              New Book
            </Link>
          </div>
          {allBooks.length === 0 ? (
            <div className="flex flex-col items-center justify-center rounded-lg border border-dashed border-gray-300 py-16">
              <p className="text-sm text-gray-500">No books yet</p>
              <Link
                href="/dashboard/books/new"
                className="mt-2 text-sm font-medium text-gray-900 hover:underline"
              >
                Create your first book
              </Link>
            </div>
          ) : (
            <div className="mt-6 grid grid-cols-1 gap-4 md:grid-cols-2 lg:grid-cols-3">
              {allBooks.map((book) => (
                <BookCard key={book.id} book={book} />
              ))}
            </div>
          )}
        </div>
      )
    }
    ```
    Key: Server Component — direct DB query, no client fetch. Filters out soft-deleted books (isNull(books.deletedAt)). Sorted by newest first.

    **File 2: `src/components/books/book-card.tsx`** (Client Component)
    ```typescript
    "use client"

    import Link from "next/link"
    import { format } from "date-fns"
    import { KDP_TRIM_SIZES } from "@/domain/kdp/trim-sizes"
    import { DeleteBookDialog } from "@/components/books/delete-book-dialog"
    import type { Book } from "@/domain/book/book"

    function getTrimSizeLabel(trimSizeId: string): string {
      return KDP_TRIM_SIZES.find((s) => s.id === trimSizeId)?.label ?? trimSizeId
    }

    export function BookCard({ book }: { book: Book }) {
      return (
        <div className="rounded-lg border bg-white p-4 shadow-sm">
          <div className="flex items-start justify-between">
            <div className="min-w-0 flex-1">
              <Link
                href={`/dashboard/books/${book.id}`}
                className="text-base font-semibold text-gray-900 hover:underline"
              >
                {book.title}
              </Link>
              {book.description && (
                <p className="mt-1 text-sm text-gray-500 line-clamp-2">{book.description}</p>
              )}
            </div>
            <DeleteBookDialog bookId={book.id} bookTitle={book.title} />
          </div>
          <div className="mt-3 flex items-center gap-3 text-xs text-gray-400">
            <span>{getTrimSizeLabel(book.trimSizeId)}</span>
            <span>·</span>
            <span>{format(new Date(book.createdAt), "MMM d, yyyy")}</span>
          </div>
        </div>
      )
    }
    ```
    Key: `"use client"` — uses useState for delete dialog. Imports KDP_TRIM_SIZES for trim size label lookup (NOT hardcoded). Uses date-fns for date formatting. Line-clamp for description truncation.

    **File 3: `src/app/(dashboard)/books/new/page.tsx`** (Client Component)
    ```typescript
    "use client"

    import { useRouter } from "next/navigation"
    import { useActionState } from "react"
    import { createBook } from "@/app/(dashboard)/books/[bookId]/actions"
    import { Header } from "@/components/layout/header"

    export default function NewBookPage() {
      const router = useRouter()
      const [state, formAction, isPending] = useActionState(
        async (_prev: { error?: string; id?: string }, formData: FormData) => {
          try {
            const result = await createBook(formData)
            router.push(`/dashboard/books/${result.id}`)
            return { id: result.id }
          } catch (error) {
            const message = error instanceof Error ? error.message : "Failed to create book"
            return { error: message }
          }
        },
        {}
      )

      return (
        <div className="max-w-lg">
          <Header title="New Book" description="Create a new book project" />
          {state.error && (
            <div className="mb-4 rounded-md bg-red-50 p-3 text-sm text-red-700">
              {state.error}
            </div>
          )}
          <form action={formAction} className="space-y-4">
            <div>
              <label htmlFor="title" className="block text-sm font-medium text-gray-700">
                Title <span className="text-red-500">*</span>
              </label>
              <input
                type="text"
                id="title"
                name="title"
                required
                maxLength={200}
                placeholder="Enter book title"
                className="mt-1 block w-full rounded-md border border-gray-300 px-3 py-2 text-sm shadow-sm focus:border-gray-500 focus:outline-none focus:ring-1 focus:ring-gray-500"
                autoFocus
              />
            </div>
            <div>
              <label htmlFor="description" className="block text-sm font-medium text-gray-700">
                Description
              </label>
              <textarea
                id="description"
                name="description"
                maxLength={2000}
                rows={3}
                placeholder="Optional description"
                className="mt-1 block w-full rounded-md border border-gray-300 px-3 py-2 text-sm shadow-sm focus:border-gray-500 focus:outline-none focus:ring-1 focus:ring-gray-500"
              />
            </div>
            <div className="flex justify-end gap-3">
              <button
                type="button"
                onClick={() => router.push("/dashboard")}
                className="rounded-md border border-gray-300 bg-white px-4 py-2 text-sm font-medium text-gray-700 hover:bg-gray-50"
              >
                Cancel
              </button>
              <button
                type="submit"
                disabled={isPending}
                className="inline-flex items-center rounded-md bg-gray-900 px-4 py-2 text-sm font-medium text-white hover:bg-gray-800 disabled:opacity-50"

                {isPending ? "Creating..." : "Create Book"}
              </button>
            </div>
          </form>
        </div>
      )
    }
    ```
    Key: Uses `useActionState` (Next.js 15 pattern) for progressive enhancement. On success, redirects to the book's edit page. KDP options use defaults (set by the Server Action). Only title is required in the create form.

    **File 4: `src/components/books/delete-book-dialog.tsx`** (Client Component)
    ```typescript
    "use client"

    import { useState } from "react"
    import { Trash2 } from "lucide-react"
    import { deleteBook } from "@/app/(dashboard)/books/[bookId]/actions"
    import { toast } from "sonner"

    export function DeleteBookDialog({ bookId, bookTitle }: { bookId: string; bookTitle: string }) {
      const [isOpen, setIsOpen] = useState(false)
      const [isDeleting, setIsDeleting] = useState(false)

      async function handleDelete() {
        setIsDeleting(true)
        try {
          await deleteBook(bookId)
          toast.success(`"${bookTitle}" has been deleted`)
        } catch {
          toast.error("Failed to delete book")
        } finally {
          setIsDeleting(false)
          setIsOpen(false)
        }
      }

      return (
        <>
          <button
            type="button"
            onClick={() => setIsOpen(true)}
            className="ml-2 rounded-md p-1 text-gray-400 hover:bg-gray-100 hover:text-gray-600"
            aria-label={`Delete "${bookTitle}"`}
          >
            <Trash2 className="h-4 w-4" />
          </button>
          {isOpen && (
            <div className="fixed inset-0 z-50 flex items-center justify-center">
              <div className="fixed inset-0 bg-black/20" onClick={() => setIsOpen(false)} />
              <div className="relative z-10 w-full max-w-sm rounded-lg bg-white p-6 shadow-lg">
                <h3 className="text-lg font-semibold">Delete Book</h3>
                <p className="mt-2 text-sm text-gray-600">
                  Are you sure you want to delete "{bookTitle}"? This action can be undone.
                </p>
                <div className="mt-4 flex justify-end gap-3">
                  <button
                    type="button"
                    onClick={() => setIsOpen(false)}
                    className="rounded-md border border-gray-300 bg-white px-4 py-2 text-sm font-medium text-gray-700 hover:bg-gray-50"
                  >
                    Cancel
                  </button>
                  <button
                    type="button"
                    onClick={handleDelete}
                    disabled={isDeleting}
                    className="rounded-md bg-red-600 px-4 py-2 text-sm font-medium text-white hover:bg-red-700 disabled:opacity-50"
                  >
                    {isDeleting ? "Deleting..." : "Delete"}
                  </button>
                </div>
              </div>
            </div>
          )}
        </>
      )
    }
    ```
    Key: Uses sonner for toast notifications on success/error. Portal-style modal overlay. Soft delete — message says "can be undone" since we use deletedAt.
