    **File 2: `src/components/books/book-settings-form.tsx`** (Client Component)
    ```typescript
    "use client"

    import { useActionState } from "react"
    import { useRouter } from "next/navigation"
    import { updateBook } from "@/app/(dashboard)/books/[bookId]/actions"
    import { BookForm } from "@/components/books/book-form"
    import { KdpOptionsForm } from "@/components/books/kdp-options-form"
    import { toast } from "sonner"

    interface BookSettingsFormProps {
      bookId: string
      defaultTitle: string
      defaultDescription?: string
      defaultTrimSizeId: string
      defaultPaperType: string
      defaultInkType: string
      defaultCoverFinish: string
    }

    export function BookSettingsForm({
      bookId,
      defaultTitle,
      defaultDescription,
      defaultTrimSizeId,
      defaultPaperType,
      defaultInkType,
      defaultCoverFinish,
    }: BookSettingsFormProps) {
      const router = useRouter()
      const [state, formAction, isPending] = useActionState(
        async (_prev: { error?: string }, formData: FormData) => {
          try {
            await updateBook(bookId, formData)
            toast.success("Book updated successfully")
            router.refresh()
          } catch (error) {
            const message = error instanceof Error ? error.message : "Failed to update book"
            return { error: message }
          }
        },
        {}
      )

      return (
        <form action={formAction} className="space-y-8">
          {state.error && (
            <div className="rounded-md bg-red-50 p-3 text-sm text-red-700">
              {state.error}
            </div>
          )}

          <div>
            <h2 className="mb-3 text-sm font-semibold uppercase tracking-wide text-gray-500">
              Book Details
            </h2>
            <div className="space-y-4">
              <div>
                <label htmlFor="title" className="block text-sm font-medium text-gray-700">
                  Title <span className="text-red-500">*</span>
                </label>
                <input
                  type="text"
                  id="title"
                  name="title"
                  required
                  maxLength={200}
                  defaultValue={defaultTitle}
                  className="mt-1 block w-full rounded-md border border-gray-300 px-3 py-2 text-sm shadow-sm focus:border-gray-500 focus:outline-none focus:ring-1 focus:ring-gray-500"
                />
              </div>
              <div>
                <label htmlFor="description" className="block text-sm font-medium text-gray-700">
                  Description
                </label>
                <textarea
                  id="description"
                  name="description"
                  maxLength={2000}
                  rows={3}
                  defaultValue={defaultDescription ?? ""}
                  className="mt-1 block w-full rounded-md border border-gray-300 px-3 py-2 text-sm shadow-sm focus:border-gray-500 focus:outline-none focus:ring-1 focus:ring-gray-500"
                />
              </div>
            </div>
          </div>

          <div>
            <h2 className="mb-3 text-sm font-semibold uppercase tracking-wide text-gray-500">
              Printing Options
            </h2>
            <KdpOptionsForm
              defaultTrimSizeId={defaultTrimSizeId}
              defaultPaperType={defaultPaperType}
              defaultInkType={defaultInkType}
              defaultCoverFinish={defaultCoverFinish}
            />
          </div>

          <div className="flex justify-end">
            <button
              type="submit"
              disabled={isPending}
              className="inline-flex items-center rounded-md bg-gray-900 px-4 py-2 text-sm font-medium text-white hover:bg-gray-800 disabled:opacity-50"
            >
              {isPending ? "Saving..." : "Save Changes"}
            </button>
          </div>
        </form>
      )
    }
    ```
    Key: Composes KdpOptionsForm for the KDP options section. Uses `useActionState` for Server Action integration. Calls `router.refresh()` to update Server Component data. Uses sonner toast for success feedback. Sections separated by headers ("Book Details" / "Printing Options").

    **File 3: `src/app/(dashboard)/books/[bookId]/page.tsx`** (Server Component)
    ```typescript
    import { notFound } from "next/navigation"
    import { eq, and, isNull } from "drizzle-orm"
    import { db } from "@/infrastructure/db/client"
    import { books } from "@/infrastructure/db/schema/books"
    import { Header } from "@/components/layout/header"
    import { BookSettingsForm } from "@/components/books/book-settings-form"

    export default async function BookSettingsPage({
      params,
    }: {
      params: Promise<{ bookId: string }>
    }) {
      const { bookId } = await params
      const book = await db
        .select()
        .from(books)
        .where(and(eq(books.id, bookId), isNull(books.deletedAt)))
        .then((rows) => rows[0])

      if (!book) {
        notFound()
      }

      return (
        <div className="max-w-2xl">
          <Header
            title={book.title}
            description={`Created ${book.createdAt}`}
          />
          <BookSettingsForm
            bookId={book.id}
            defaultTitle={book.title}
            defaultDescription={book.description ?? undefined}
            defaultTrimSizeId={book.trimSizeId}
            defaultPaperType={book.paperType}
            defaultInkType={book.inkType}
            defaultCoverFinish={book.coverFinish}
          />
        </div>
      )
    }
    ```
    Key: Server Component — fetches book from DB, passes to client BookSettingsForm. Uses Next.js 15 `params: Promise` pattern (async params). Calls `notFound()` if book doesn't exist or is soft-deleted. Filters by `isNull(books.deletedAt)`.
